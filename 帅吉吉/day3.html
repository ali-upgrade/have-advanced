    # 浮动会产生的问题

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			*{
				margin: 0;
				padding: 0;
			}
			ul{
				list-style: none;
				/* 当一个元素中的子元素都浮动，则会导致父元素高度为0的问题(高度坍塌) */
				/* 1. 直接给父元素加高度 */
				/* height: 200px; */
				/* 2. 让父元素也浮动起来，并且给后续元素设置清除浮动*/
				/* float: left; */
				/* 3. 固定用法 - 溢出隐藏 - 不能用在下拉框功能的实现中 */
				overflow: hidden;
			}
			/* 4. 开发中常用方式，使用伪类的方式,固定用法*/
			ul:after{
				content: "";
				display: block;
				clear: both;
			}
			li{
				width: 200px;
				height: 200px;
				float: left;
			}
			.a1{
				background-color: red;
			}
			.a2{
				background-color: green;
			}
			.a3{
				background-color: blue;
			}
			div{
				width: 200px;
				height: 200px;
				background-color: black;
				/* 清除浮动 */
				/* clear: none/left/right/both; */
				/* 清除两侧浮动 */
				/* clear: both; */
			}
		</style>
	</head>
	<body>
		<ul>
			<li class="a1"></li>
			<li class="a2"></li>
			<li class="a3"></li>
		</ul>
		<div></div>
	</body>
</html>

    # 导航栏

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			*{
				margin: 0;
				padding: 0;
			}
			ul{
				list-style: none;
				float: left;
			}
			.nav-all{
				width: 200px;
				height: 40px;
				background-color: red;
				color: white;
				/* 行高 - 控制块级元素中内容垂直居中 - 取值为当前元素高度即可 */
				line-height: 40px;
				float: left;
			}
			li{
				width: 100px;
				height: 40px;
				background-color: green;
				text-align: center;
				line-height: 40px;
				float: left;
			}
			li:hover{
				background-color: red;
			}
			a{
				color: white;
				text-decoration: none;
			}
		</style>
	</head>
	<body>
		
		<div class="nav">
			<div class="nav-all">&nbsp;&nbsp;全部商品分类</div>
			<ul>
				<li><a href="#">首页</a></li>
				<li><a href="#">商城</a></li>
				<li><a href="#">店铺</a></li>
			</ul>
		</div>
		
	</body>
</html>

    # 尺寸与单元

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			html{
				font-size: 10px;
			}
			h1{
				font-size: 20px;
			}
			div{
				width: 10rem;
				/* 尺寸单位
					px 像素
					%  百分比(相对于父元素)
					mm 毫米
					cm 厘米
					in 英寸
					pt 磅
					em 相对单位(相对于父元素)
					rem 相对单位(相对于根元素html)
				 */
				height: 200px;
				background-color: #EA2000;
				/* 颜色单位 - 合法的颜色值
					1. 直接写英文单词
					2. #rrggbb; #六位十六进制数字
						0-9 a-f
						如果rrggbb的值两两一样，可以简写为#rgb
						#aabbcc == #abc
						#000000 = #000 黑色
						#fff 白色
						#f00 红色
						#0f0 绿色
						#00f 蓝色
						如果rgb的值相同，则是灰色，越接近0，颜色越深，越接近f，颜色越浅
						#333
						#666
						#999
						#aaa
						#ccc
						
				 */
			}
		</style>
	</head>
	<body>
		<h1>
			<div></div>
		</h1>
		
	</body>
</html>
