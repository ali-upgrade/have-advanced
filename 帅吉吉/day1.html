    # 基本结构

    <!-- 1. 声明文档类型，告诉浏览器我们的代码时HTML的那个版本 -->
<!-- h5的声明文档类型的方式 -->
<!-- 单标记 -->
<!DOCTYPE html>
<!-- 双标记 -->
<!-- 2. 网页的部分 -->
<html>
	<!-- 2.1 网页的头部-声明网页头信息 -->
	<head>
		<!-- 设置字符源码集格式 - 解决乱码  -->
		<meta charset="utf-8">
		<title>网页的标题</title>
	</head>
	<!-- 2.2 网页的主体内容部分 -->
	<body>
		<!-- 只要想是现在网页中的任何内容，都需要写在body中 -->
	</body>
</html>

    # 使用CSS的三种方式

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<!-- 2. 内部样式表 -->
		<!-- 优势 - 不需要新建文件、也能同时控制多行样式，适用于平时的练习 -->
		<!-- 缺点 - 没有结构与样式分离，不适合项目 -->
		<style>
			/* 选择器{
				属性名: 属性值;
			} */
			/* 选择页面中的i标签 */
			i{
				color: green;
			}
		</style>
		<!-- 3. 外部样式表 -->
		<!-- 优势 - 结构与样式分离，便于维护 -->
		<!-- 缺点 - 如果代码量很少，比较麻烦,不适用于平时的练习 -->
		<!-- <link rel="stylesheet" href="要引用的外部样式表的路径"> -->
		<link rel="stylesheet" href="test.css">
	</head>
	<body>
		<!-- Ctrl+/ -->
		<!-- CSS - 层叠样式表/级联样式表/样式表 -->
		<!-- 1. 内联样式表 -->
		<!-- 优势 - 直观 - 优先级最高 - JS中都是以内联的方式添加样式 -->
		<!-- 缺点 - 不利于维护，没有结构与样式分离 -->
		<b style="color: red;">加粗</b>
		<i>倾斜</i>
		<u>下划线</u>
	</body>
</html>

    <!DOCTYPE html>

	<head>
		<meta charset="utf-8">
		<title></title>
	</head>
	<body>
		<!-- 段落元素 p ，在网页中使用的频率较高-->
		<!-- 特点
			1. 独占一行
			2. 上下行间距
			3. P标签一般用于网页中的纯文本
			4. 注意: P标签中不能嵌套其他的块级元素
		 -->
		帅吉吉
		<p>这是P1</p>
		<p>这是P2</p>
		<!-- 标题元素 -->
		<!-- 特点
			1. 文本加粗
			2. 文本尺寸依次减小
			3. 独占一行
			4. 上下行间距
		 -->
		<h1>这是H1</h1>
		<h2>这是H2</h2>
		<h3>这是H3</h3>
		帅吉吉
		<h4>这是H4</h4>
		<H5>这是H5</H5>
		<H6>这是H6</H6>
	</body>
</html>

    # 块级元素和行内元素

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			b{
				/* 显示为块级元素 */
				display: block;
			}
			p{
				/* 显示为行内元素 */
				display: inline;
			}
		</style>
	</head>
	<body>
		<b>加粗</b>
		<i>倾斜</i>
		<p>段落</p>
		<!-- 块级元素 - 在默认情况下，可以独占一行的元素 -->
			<!-- h1 - h6   p   div(没有任何默认样式的块级元素) -->
		<!-- 行内元素 - 在默认情况下，可以和其他行内元素在一行显示 -->
			<!-- b i s u sub sup span(没有任何默认样式的行内元素) -->
	</body>
</html>

    # Css中的常用的样式

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			div{
				/* 1. 文本颜色 */
				color: red;
				/* 2. 背景颜色 */
				background-color: green;
				/* 3. 文本尺寸 */
				font-size: 12px;
				/* 4. 文本加粗 */
				/* font-weight: bold/100/200.../900; */
				/* 5. 文本修饰 - 是否加1下划线 */
				text-decoration: none;
				/* 6. 字体 */
				font-family: "宋体";
				/* 7. 宽度 */
				width: 200px;
				/* 8. 高度 */
				height: 200px;
				/* 9. 文本排列 */
				/* text-align: left/center/right; */
				text-align: center;
			}
			span{
				/* 1. 文本颜色 */
				color: red;
				/* 2. 背景颜色 */
				background-color: green;
				/* 3. 文本尺寸 */
				font-size: 12px;
				/* 4. 文本加粗 */
				/* font-weight: bold/100/200.../900; */
				/* 5. 文本修饰 - 是否加下划线 */
				text-decoration: underline;
				/* 6. 字体 */
				font-family: "幼圆";
				/* 7. 宽度 */
				width: 200px;
				/* 8. 高度 */
				height: 200px;
				/* 9. 文本排列 */
				/* text-align: left/center/right; */
				text-align: center;
			}
			/* 注意: 块级元素可以设置尺寸，行内元素设置的尺寸会失效 */
			/* 块级元素一般用来做布局 */
			/* 行内元素一般用来做网页中的细微的控制 */
			/* 居中的前提是元素中的内容宽度要比元素本身的宽度小 */
			/* 行内元素不能设置内容居中 */
		</style>
	</head>
	<body>
		<div>块级元素</div>
		<span>行内元素</span>
	</body>
</html>

    # 文本标记练习

    <!DOCTYPE html>
<html>
	<head>
		<meta charset="utf-8">
		<title></title>
		<style>
			h1{
				text-align: center;
			}
			span{
				color: red;
			}
			img{
				width: 200px;
			}
			div{
				text-align: center;
			}
		</style>
	</head>
	<body>
		<!-- 实体(实业)字符 -->
		<!-- &lt; < -->
		<!-- &gt; > -->
		<!-- &nbsp; 空格 -->
		<!-- &copy; © -->
		<!-- &yen; ¥ -->
		<!-- 在网页中编写的空格和换行，最终都会被解析成是一个空格 -->
		<h1>HTML5<span>&lt;DAY01&gt;</span></h1>
		<hr>
		<h2 id="test">1 HTML 文档片段</h2>
		<h3>1.1 问题</h3>
		<p>创建如图-1所示的HTML页面: </p>
		<div>
			<img src="../images/dlrb.jpeg" alt="">
		</div>
		<div>图-1</div>
		
		
		<h3>1.2 方案</h3>
		<p>使用HTML标记来完成该页面效果:</p>
		<h3>1.3 实现</h3>
		<p>Lorem ipsum dolor sit amet consectetur, adipisicing elit. Maxime dolorum quia</p>
		<p>Lorem ipsum dolor sit amet consectetur adipisicing elit. Cumque so</p>
		<!-- pre - 预格式化 - 保留源文件中的空格和换行 -->
		<pre>
			&lt;p align="center" title="段落"&gt;
				居中显示的段落
			&lt;/p&gt;
		</pre>
		
	</body>
</html>
